
############# Créer la table des objets #############

CREATE TABLE objets (
	id INT PRIMARY KEY,
    objet_nom VARCHAR(255),
    zoom INT,
    icone VARCHAR(255),
    bloque INT,
texte TEXT,
	code VARCHAR(255),
	recuperable boolean,
	remove boolean,
	coord GEOMETRY(POINT, 4326),
	showtext boolean
);



############# Créer la table des scores #############

CREATE TABLE score (
	nom VARCHAR(255) NOT NULL,
	temps TIME WITHOUT TIME ZONE
);
